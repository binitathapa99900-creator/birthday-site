    <script>
        function triggerSurprise() {
            // Confetti effect
            confetti({
                particleCount: 150,
                spread: 70,
                origin: { y: 0.6 },
                colors: ['#ffafcc', '#023e8a', '#ffffff']
            });

            document.getElementById('surpriseBox').style.display = 'block';
            document.getElementById('cakeBox').style.display = 'none';
        }

        function toggleCake() {
            var box = document.getElementById('cakeBox');
            box.style.display = (box.style.display === 'block') ? 'none' : 'block';
            document.getElementById('surpriseBox').style.display = 'none';
        }

        function toggleMusic() {
            var song = document.getElementById('bdaySong');
            var text = document.getElementById('musicText');
            if (song.paused) {
                song.play();
                text.innerText = "Pause Song";
            } else {
                song.pause();
                text.innerText = "Play Song";
            }
        }
    </script>
